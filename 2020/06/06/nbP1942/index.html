const p=1000000007;
var
  a,b,s:array[0..1000000] of int64;
  ans:int64;
  z,n,i,tot:longint;
procedure sort (l,h:longint);
    var
        i,j,t,m:longint;
    begin
        i:=l; j:=h; m:=a[(i+j)>>1];
        repeat
            while a[i]<m do inc(i);
	        while m<a[j] do dec(j);
	        if i<=j then begin
	            t:=a[i]; a[i]:=a[j]; a[j]:=t;
		        inc(i); dec(j);
	        end;
        until i>j;
        if i<h then sort(i,h);
        if j>l then sort(l,j);
    end;
begin
  read(n);
  for i:=1 to n do read(a[i]);
  sort(1,n); z:=1;
  for i:=2 to n+1 do
    if a[i]=a[i-1] then z:=z+1
	  else begin
	    if z>1 then begin
		  tot:=tot+1;
		  b[tot]:=(z*(z-1)) mod p;
		  s[tot]:=(s[tot-1]+b[tot]) mod p;
		end;
	    z:=1;
	  end;
  for i:=2 to tot do ans:=(ans+(b[i]*s[i-1]) mod p) mod p;
  write(ans);
end.
