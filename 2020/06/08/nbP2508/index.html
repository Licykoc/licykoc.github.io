总结一下题意就是拿一个n*n的矩阵进行若干个操作可不可以变成目标矩阵
水是真的水，但是烦也是真的烦,还是看代码吧（防止抄袭，只展示部分）

function is:boolean; //is函数用来判断当前矩阵和目标矩阵相不相同
  var
    i,j:longint;
  begin
    for i:=1 to n do 
	  for j:=1 to n do 
	    if t[i,j]<>map[i,j] then exit(false);
		//t数组表示当前矩阵，map数组表示目标数组
	exit(true);
  end;

procedure g1;
//第#1种交换
  var
    i,j:longint;
  begin
    for i:=1 to n do 
	  for j:=1 to n do 
	    t[j,n-i+1]:=a[i,j];
		//a数组表示原始矩阵,t数组见上
  end;

procedure g2; 
//第#2种交换
  var
    i,j:longint;
  begin
    for i:=1 to n do 
	  for j:=1 to n do 
	    t[n-i+1,n-j+1]:=a[i,j];
  end;

procedure g3;
//第#3种交换
  var
    i,j:longint;
  begin
    for i:=1 to n do 
	  for j:=1 to n do 
	    t[n-j+1,i]:=a[i,j];
  end;

procedure g4;
//第#4种交换

//第#5种交换
